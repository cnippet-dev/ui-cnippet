## Usage

```tsx
import { ScrollReveal } from "@/components/motion/scroll-reveal";
```

```tsx
<ScrollReveal
  variants={{ hidden: { opacity: 0, y: 24 }, visible: { opacity: 1, y: 0 } }}
  transition={{ duration: 0.5, ease: "easeOut" }}
>
  <YourContent />
</ScrollReveal>
```

## Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `children` | `ReactNode` | — | Content to reveal |
| `variants` | `{ hidden: Variant; visible: Variant }` | `{ hidden: { opacity: 0 }, visible: { opacity: 1 } }` | Motion variants for the hidden and visible states |
| `transition` | `Transition` | — | Transition applied to the visible state |
| `viewOptions` | `UseInViewOptions` | — | Options passed to Motion's `useInView` (e.g. `amount`, `margin`) |
| `as` | `React.ElementType` | `"div"` | HTML element or component to render as |
| `once` | `boolean` | — | When `true`, the element stays visible after the first reveal and never hides again |
