## Usage

```tsx
import {
  FloatingElement,
  ParallaxFloating,
} from "@/components/motion/parallax-floating";
```

```tsx
<div className="relative h-96 w-full overflow-hidden">
  <ParallaxFloating sensitivity={1}>
    <FloatingElement className="left-[10%] top-[20%]" depth={0.5}>
      <div className="h-10 w-10 rounded-xl bg-violet-500/20" />
    </FloatingElement>
    <FloatingElement className="right-[15%] top-[15%]" depth={2}>
      <div className="h-6 w-6 rounded-full bg-pink-500/30" />
    </FloatingElement>
  </ParallaxFloating>
  <div className="relative z-10">Your content here</div>
</div>
```
