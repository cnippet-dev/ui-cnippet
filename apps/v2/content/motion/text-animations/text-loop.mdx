## Usage

```tsx
import { TextLoop } from "@/components/motion/text-loop";
```

```tsx
<TextLoop interval={2}>
  <span>First</span>
  <span>Second</span>
  <span>Third</span>
</TextLoop>
```

## Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `children` | `ReactNode[]` | — | Array of items to loop through |
| `interval` | `number` | `2` | Seconds between transitions |
| `transition` | `Transition` | `{ duration: 0.3 }` | Motion transition for enter/exit |
| `variants` | `Variants` | fade + slide-y | Custom enter/exit variants |
| `mode` | `AnimatePresenceProps["mode"]` | `"popLayout"` | AnimatePresence mode |
| `trigger` | `boolean` | `true` | Set `false` to pause the loop |
| `onIndexChange` | `(index: number) => void` | — | Callback fired on each step |
| `className` | `string` | — | Class applied to the container |
